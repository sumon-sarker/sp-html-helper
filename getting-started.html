Getting Started

Basic HTML

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="utf-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <!-- SpHtmlHelper CSS -->
    <link rel="stylesheet" type="text/css" href="dist/css/sp-html-helper.css">
  </head>
  <body>
  	<h1>Welcome to SpHtmlHelper Script</h1>


  	<!-- SpHtmlHelper Javascript -->
  	<script type="text/javascript" src="dist/js/sp-html-helper.js"></script>
  	<script type="text/javascript" src="dist/js/sp-html-helper-config.js"></script>
  </body>
</html>

Required META Tags

<meta charset="utf-8">
<meta http-equiv="X-UA-Compatible" content="IE=edge">
<meta name="viewport" content="width=device-width, initial-scale=1">

Required CSS Stylesheet

<link rel="stylesheet" type="text/css" href="dist/css/sp-html-helper.css">

Required SpHtmlHelper Scripts

<script type="text/javascript" src="dist/js/sp-html-helper.js"></script>
<script type="text/javascript" src="dist/js/sp-html-helper-config.js"></script>

Required Configuration : dist/js/sp-html-helper-config.js

var SP = new SpHtmlHelper({
	debugMode 	: false,	/*If debugMode:true, You can see result in your console. This is most important thing about Development mode*/
	screenSize 	: 768		/*You can define window screenSize, SpHtmlHelper scripts depended on screenSize*/
});
