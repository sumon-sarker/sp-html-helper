Getting Started

Basic HTML

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="utf-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <!-- SpHtmlHelper CSS -->
    <link rel="stylesheet" type="text/css" href="dist/css/sp-html-helper.css">
  </head>
  <body>
  	<h1>Welcome to SpHtmlHelper Script</h1>


  	<!-- SpHtmlHelper Javascript -->
  	<script type="text/javascript" src="dist/js/sp-html-helper.js"></script>
  	<script type="text/javascript" src="dist/js/sp-html-helper-config.js"></script>
  </body>
</html>

Configuration : dist/js/sp-html-helper-config.js

var SP = new SpHtmlHelper({
	debugMode 	: false,
	screenSize 	: 768
});

Menu Configuration

Configuration : dist/js/sp-html-helper-config.js

SP.MenuConfig({
	multiSiteMenu	: false,
	languageString 	: '/en/',
	copyTargetMenu 	: {
		copy 		: false,
		targetTag 	: ""
	}
});

Add Menu

Configuration : dist/js/sp-html-helper-config.js

Sp.AddMenu({
	menuLogo 		: 'dist/img/sp-html-helper.png',
	targetContainer : 'body',
	menuItems 		: [
		{
			title 		: 'SpHtmlHelper1',
			link 		: '#1',
			css_class 	: ''
		},
		{
			title 		: 'SpHtmlHelper2',
			link 		: '#2',
			css_class 	: ''
		},
	]
});
